The Initializer is the first thing that runs on both the server and client. It concurrently requires every ModuleScript in a folder using task.spawn, waits for all of them to finish, then calls Init and Start in order so modules that depend on each other don't race. It also tracks any module that takes longer than a second to load and prints a full performance report when done, including branch, place version, and which modules were slow.

ActorNetwork handles everything to do with actors on the client side — creation, destruction, ability icons, cooldown rendering, and input. The ability icons are pulled from an object pool instead of Instance.new every time, so spawning in as a new character doesn't tank performance. On cleanup it returns all icons back to the pool and disconnects every connection cleanly.

Jason is one of the recoded killers. Every ability runs dual-context: server handles hitboxes, damage, status effects, and stat tracking. Client handles animations, speed multipliers, and visual effects. RagingPace is his enrage ability — it locks his stamina cap, reveals the nearest survivor's aura, tweens his FOV and speed, and auto-pacifies after a timer.

Two separate systems in one file. The head follow lerp offsets the camera to track the Head bone instead of staying locked to HumanoidRootPart, which makes third-person feel a lot more alive. The shakey cam binds to RenderStep at Camera priority and oscillates the camera CFrame using sin/cos when health drops below 50%, with intensity scaling linearly down to zero at full health. Both are toggleable through player settings.

Custom networking layer that replaces all direct RemoteEvent usage across the codebase. Everything is packed into a binary buffer before sending and unpacked on the other side, which keeps bandwidth low. There's a per-player rate limiter on the server that drops packets if a client sends too much in one tick. SetSignal creates one-shot listeners that auto-remove themselves after the first matching packet.

type SlowModule = {
    Path: string,
    Time: number,
    Phase: string,
}

type ModuleLoadResult = {
    Module: ModuleScript,
    Success: boolean,
    Result: any?,
    LoadTime: number,
}

local Initializer = {}
Initializer.__index = Initializer

local TotalModulesLoaded: number = 0
local SlowModules: {SlowModule} = {}
local SLOW_THRESHOLD: number = 1.0

local function loadModulesConcurrently(folder: Folder): {ModuleLoadResult}
    local results: {ModuleLoadResult} = {}
    local pendingCount = 0
    local completedCount = 0

    local function countModules(f: Folder): number
        local count = 0
        for _, child in f:GetChildren() do
            if child:GetAttribute("NoInitializing") then continue end
            if child:IsA("Folder") then count += countModules(child)
            elseif child:IsA("ModuleScript") then count += 1 end
        end
        return count
    end

    pendingCount = countModules(folder)

    local function loadRecursive(currentFolder: Folder)
        for _, child in currentFolder:GetChildren() do
            if child:GetAttribute("NoInitializing") then continue end
            if child:IsA("Folder") then
                loadRecursive(child)
            elseif child:IsA("ModuleScript") then
                task.spawn(function()
                    local startTime = os.clock()
                    local success, result = pcall(require, child)
                    local loadTime = os.clock() - startTime

                    table.insert(results, {
                        Module   = child,
                        Success  = success,
                        Result   = result,
                        LoadTime = loadTime,
                    })

                    if loadTime > SLOW_THRESHOLD then
                        table.insert(SlowModules, {
                            Path  = child:GetFullName(),
                            Time  = loadTime,
                            Phase = "Load",
                        })
                    end

                    completedCount += 1
                end)
            end
        end
    end

    loadRecursive(folder)

    while completedCount < pendingCount do task.wait() end

    return results
end

local function callMethodSequential(directory: string, methodname: string)
    for moduleName, requiredModule in Initializer[directory] do
        if not requiredModule[methodname] then continue end

        local startTime = os.clock()
        pcallMethod(requiredModule, methodname)
        local callTime = os.clock() - startTime

        if callTime > SLOW_THRESHOLD then
            table.insert(SlowModules, {
                Path  = moduleName,
                Time  = callTime,
                Phase = methodname,
            })
        end
    end
end

    Initializer.Modules = {}

            Initializer:loadModules(characterSystems, "CharacterSystems")
        end)
    end

    Initializer.Loaded = true
    if isClient then shared.InitializerFinished = true end

function ActorNetwork:Start()

    Network:SetConnection("ActorCreated", "REMOTE_EVENT", function(actorInfo)

        local skinFolder = Actors:ApplySkinDataToActorInfo(actorInfo)
        actorInfo.CustomBehavior = skinFolder
            and skinFolder:FindFirstChild("Behavior")
            and require(skinFolder.Behavior) or {}

        Actors.CurrentActors[actorInfo.Player] = actorInfo
        Util:PreloadAssets(actorInfo.Config)

        -- [ ... chase theme + created callback threads omitted ... ]

        task.spawn(function()
            if actorInfo.Player ~= player then return end

            ActorNetwork:Destroy()
            actorInfo.Animations = {}
            actorInfo.Cooldowns  = {}

            -- [ ... animation loading + movement anim handling omitted ... ]

            local hotbar = game.ReplicatedStorage.Systems.Player.UI.CustomBackpack.Backpack:Clone()
            hotbar.Name   = "AbilityContainer"
            hotbar.Parent = player.PlayerGui.MainUI
            ActorNetwork.AbilityHotbar = hotbar

            local abilities = actorInfo.Behavior.Abilities or {}
            for abilityName, abilityData in pairs(actorInfo.CustomBehavior.Abilities or {}) do
                abilities[abilityName] = abilityData
            end

            -- initialize ability icon pool if it doesn't exist yet
            if not ObjectPool.Pools["AbilityIcons"] then
                ObjectPool:CreatePool("AbilityIcons",
                    game.ReplicatedStorage.Assets.UI.AbilityIcon, 20)
            end

            actorInfo.PooledIcons = {}

            for abilityName, abilityInfo in pairs(abilities or {}) do

                -- get icon from pool instead of Instance.new every time
                local abilityIcon = ObjectPool:Get("AbilityIcons")
                if not abilityIcon then
                    warn(`[ActorNetwork]: pool miss on AbilityIcons for {abilityName}`)
                    continue
                end

                table.insert(actorInfo.PooledIcons, {
                    Name     = abilityName,
                    Instance = abilityIcon,
                })

                abilityIcon.Image          = abilityInfo.Icon or "rbxassetid://..."
                abilityIcon.LayoutOrder    = abilityInfo.LayoutOrder
                abilityIcon.AbilityName.Text = abilityInfo.Name
                    or abilityName:gsub("(%l)(%u)", "%1 %2")
                abilityIcon.Name    = abilityName
                abilityIcon.Parent  = hotbar
                abilityIcon.Visible = true

                -- [ ... keybind connections, hold detection omitted ... ]
            end
        end)

        task.defer(Util.FixCharacter, Util, actorInfo.Rig, `{actorInfo.ActorType}s`)
        Actors[`__actorCreated`]:Fire(actorInfo)
    end)

    -- [ ... ActorDestroyed, UpdateAbilityCharges, UpdateAbilityCooldown,
    --       UseActorAbility connections omitted ... ]

end

function ActorNetwork:Destroy()
    -- return pooled icons before wiping the hotbar
    local actorInfo = require(game.ReplicatedStorage.Modules.Actors).CurrentActors[player]
    if actorInfo and actorInfo.PooledIcons then
        for _, iconData in pairs(actorInfo.PooledIcons) do
            ObjectPool:Return("AbilityIcons", iconData.Instance)
        end
        table.clear(actorInfo.PooledIcons)
    end

    if actorInfo then
        if actorInfo.Cooldowns     then table.clear(actorInfo.Cooldowns)     end
        if actorInfo.HeldAbilities then table.clear(actorInfo.HeldAbilities) end
    end

    if ActorNetwork.AbilityHotbar then
        ActorNetwork.AbilityHotbar:Destroy()
        ActorNetwork.AbilityHotbar = nil
    end

    for indx, connection in pairs(ActorNetwork.Connections) do
        if typeof(connection) == "RBXScriptConnection" then
            connection:Disconnect()
        end
        ActorNetwork.Connections[indx] = nil
    end
    table.clear(ActorNetwork.Connections)
end

return ActorNetwork

            {Parent = killerInfo.Rig.PrimaryPart})

        for _, obj in pairs(killerInfo.Instances.EnragedVisuals) do
            obj.Enabled = false
        end

        Network:FireClientConnection(killerInfo.Player,
            `{killerInfo.Player.Name}EnrageStateAsync`, "REMOTE_EVENT")

    elseif game["Run Service"]:IsClient() then
        killerInfo.AnimationType     = nil
        killerInfo.MovementAnimSpeed = nil
        Network:FireConnection("DisableSprinting", "ENRAGED", false)

        if killerInfo.Meta.TargetedPlayer then
            Util:RevealPlayerAura(killerInfo.TargetedPlayer,

            killerInfo.Meta.TargetedPlayer = nil

        sprinting.StaminaCap = nil
    end

local CameraHandler = {}
CameraHandler.Connections = {}

local previousMouseDelta = Vector2.new(0, 0)
local currentTurn        = 0

local function lerp(a, b, t)
    return a + (b - a) * t
end

local function smoothMouseDelta(mouseDelta: Vector2, smoothingFactor: number): Vector2
    previousMouseDelta = previousMouseDelta:Lerp(mouseDelta, smoothingFactor)
    return previousMouseDelta
end

-- makes your camera follow the head instead of locking to HRP
local function lerpCamera(deltaTime: number)
    local character = game.Players.LocalPlayer.Character
    if not character then return end

    local humanoid = character:FindFirstChild("Humanoid")
    if not humanoid then return end

    if humanoid:GetState() == Enum.HumanoidStateType.Dead
    or humanoid:GetState() == Enum.HumanoidStateType.Physics then
        humanoid.CameraOffset = Vector3.new(0, 0, 0)
        return
    end

    local hrp    = character:FindFirstChild("HumanoidRootPart")
    local head   = character:FindFirstChild("Head")
    local camera = workspace.Camera

    if not hrp or camera.CameraType == Enum.CameraType.Scriptable or not head then
        return
    end

    local targetOffset =
        (hrp.CFrame + Vector3.new(0, 1.5, 0)):PointToObjectSpace(head.Position)

    humanoid.CameraOffset =
        humanoid.CameraOffset:Lerp(targetOffset, deltaTime * 5)
end

-- dynamic camera sway based on mouse movement speed
local function updateCameraSway(deltaTime: number)
    local character = game.Players.LocalPlayer.Character
    if not character then return end

    local camera = workspace.Camera
    if camera.CameraType == Enum.CameraType.Scriptable    then return end
    if character:GetAttribute("UsingAnimations")           then return end
    if character:GetAttribute("NoCamFollow")               then return end

    local maxTurnAngle  = math.clamp(math.rad(dynamicCameraSetting.Value), 0, math.huge)
    local mouseDelta    = UserInputService:GetMouseDelta()
    local smoothedDelta = smoothMouseDelta(mouseDelta, 0.5)
    local tiltIntensity = math.clamp(smoothedDelta.Magnitude / 20, 0, maxTurnAngle)

    currentTurn = lerp(
        currentTurn,
        smoothedDelta.X > 0 and tiltIntensity or -tiltIntensity,
        3 * deltaTime
    )

    camera.CFrame = camera.CFrame * CFrame.Angles(0, 0, currentTurn)
end

-- shakey cam: camera oscillates when health drops below 50%
local maxShake            = 1.5
local oscillationFrequency = 2.5
local randomnessScale     = 0.3
local rng                 = Random.new()
local timeElapsed         = 0

local function calculateShakeIntensity(health, maxHealth)
    local threshold = maxHealth / 2
    if health >= threshold then return 0 end
    return maxShake * (1 - health / threshold)
end

game["Run Service"]:BindToRenderStep("ShakeyCamera",
    Enum.RenderPriority.Camera.Value + 10,
    function(dt)
        local humanoid = player.Character
            and player.Character:FindFirstChildOfClass("Humanoid")
        if not humanoid then return end

        local health    = humanoid.Health
        local maxHealth = humanoid.MaxHealth

        local shouldShake = health < maxHealth / 2
            and health > 0
            and screenShakeSetting.Value
            and shakeWhenLowSetting.Value

        if not shouldShake then return end

        timeElapsed += dt
        local intensity = calculateShakeIntensity(health, maxHealth)

        local offsetX = math.sin(timeElapsed * oscillationFrequency) * intensity
        local offsetY = math.cos(timeElapsed * oscillationFrequency) * intensity
        local offsetZ = math.sin(timeElapsed * oscillationFrequency * 0.5) * intensity

        offsetX = (offsetX + (rng:NextNumber() - 0.5) * randomnessScale) * intensity
        offsetY = (offsetY + (rng:NextNumber() - 0.5) * randomnessScale) * intensity
        offsetZ = (offsetZ + (rng:NextNumber() - 0.5) * randomnessScale) * intensity

        workspace.CurrentCamera.CFrame *=
            CFrame.new(offsetX * 0.1, offsetY * 0.1, offsetZ * 0.1)
    end
)

        -- [ ... player/character awaiting, connection setup omitted ... ]

end

local RemoteListener = {}

local __connections: {[string]: Connection}        = {}
local __signals:     {[string]: RBXScriptConnection} = {}

-- dispatches incoming packet to its registered handler
local function onConnection(player: Player?, connType: string, connName: string,
    buf: buffer, instances: {Instance}?): any?

    local conn = __connections[connName]
    if not conn then
        Utilities.warn(script:GetFullName(), `no handler for: {connName}`)
        return nil
    end
    if conn.Type ~= connType then return nil end

    local args = Utilities.unpack(buf, instances)
    return if player
        then conn.Callback(player, table.unpack(args))
        else conn.Callback(table.unpack(args))
end

-- server side: guard, rate check, dispatch
local function onServerPacket(connType, player, connName, buf, instances)
    if not player or not player.Parent then return end
    if typeof(buf) ~= "buffer"         then return end
    if not Utilities.checkRateLimit(player, buffer.len(buf)) then return end
    onConnection(player, connType, connName, buf, instances)
end

-- [ ... remote setup (RE, URE, RF, BE) + signal wiring omitted ... ]

function RemoteListener:SetConnection(
    connectionName: string,
    connectionType: RemoteConnection,
    callback: (Player?, ...any) -> any?): Connection | nil

    local conn: Connection = {
        Name     = connectionName,
        Type     = connectionType,
        Callback = callback,
    }
    __connections[connectionName] = conn
    return conn
end

-- one-shot signal: waits for first matching packet then self-destructs
function RemoteListener:SetSignal(signalName: string)
    local fired    = false
    local captured = {}

    RemoteListener:SetConnection(`Signal{signalName}`, "REMOTE_EVENT", function(...)
        fired    = true
        captured = table.pack(...)
        RemoteListener:RemoveConnection(`Signal{signalName}`, "REMOTE_EVENT")
    end)

    return {
        Wait = function()
            repeat game["Run Service"].Heartbeat:Wait() until fired
            return captured
        end,
    }
end

-- [ ... FireClientConnection, FireAllClientConnection,
--       FireServerConnection, FireConnection, GetRemote omitted ... ]

function RemoteListener:RemoveConnection(
    connectionName: string,
    connectionType: RemoteConnection): Connection | nil

    local conn = __connections[connectionName]
    if not conn or conn.Type ~= connectionType then return nil end
    __connections[connectionName] = nil
    return conn
end

function RemoteListener:Destroy()
    table.clear(__connections)
    for _, sig in __signals do sig:Disconnect() end
    table.clear(__signals)
end

export type Connection = {
    Name:     string,
    Type:     RemoteConnection,
    Callback: (Player?, ...any) -> any?,
}

export type RemoteConnection =
    "REMOTE_EVENT" | "REMOTE_FUNCTION" | "UREMOTE_EVENT" | "BINDABLE_EVENT"

export type RemoteListener = typeof(setmetatable({}, RemoteListener)) & {
    Connections: {[string]: Connection},
    Signals:     {[string]: RBXScriptConnection},
}

return RemoteListener
